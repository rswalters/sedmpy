[docs]def getRaFromDeg(deg):
	hh = int(deg/15)
	mm = int((deg/15. - hh)*60)
	ss = ((deg/15. - hh)*60 - mm )*60
	return hh,mm,("%.2f"%ss).zfill(4)



[docs]def getDecFromDeg(deg):
	if (deg<0):
		sign = -1
	else:
		sign = 1
	deg = np.abs(deg)
	hh = int(deg)
	mm = int((deg - hh)*60)
	ss = ((deg - hh)*60 - mm )*60
	return sign*hh,mm,("%.2f"%ss).zfill(4)
